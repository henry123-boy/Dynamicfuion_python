function(NNRT_add_test name)
    add_executable(${name} ${ARGN})
    target_link_libraries(${name} PUBLIC nnrt_cpp test_utils)
    target_link_libraries(${name} PUBLIC Python3::Python ${CATCH2_TARGET})
    add_test(NAME ${name} COMMAND ${name})
endfunction()
